<script>
function checkCredentials(){
  const user = document.getElementById('username').value;
  const pass = document.getElementById('password').value;
  if(user.trim() && pass.trim()){
    document.getElementById('loginScreen').style.display='none';
    document.getElementById('authScreen').style.display='flex';
  } else {
    alert('Please enter username and password');
  }
}
function verifyAuth(){
  const code = document.getElementById('authCode').value;
  if(/^[0-9]{6}$/.test(code)){
    document.getElementById('authScreen').style.display='none';
    document.getElementById('app').style.display='block';
    document.querySelectorAll('.section').forEach(s => s.style.display = 'none');
    document.getElementById('dashboard').style.display='block';
    if(document.getElementById('activityChart')) initChart();
    updateTimes();
  } else {
    alert('Please enter a valid 6-digit authenticator code');
  }
}
function showSection(id, el){
  document.querySelectorAll('.section').forEach(s=>s.style.display='none');
  const section = document.getElementById(id);
  if(section) section.style.display='block';
  document.querySelectorAll('nav a').forEach(a=>a.classList.remove('active'));
  if(el) el.classList.add('active');
}
function initChart(){
  const ctx = document.getElementById('activityChart');
  if(ctx) new Chart(ctx,{type:'bar',data:{labels:['Mon','Tue','Wed','Thu','Fri'],datasets:[{data:[12,19,8,15,5]}]},options:{plugins:{legend:{display:false}}}});
}
function updateTimes(){
  const now = new Date();
  const t1 = document.getElementById('t1');
  const t2 = document.getElementById('t2');
  const t3 = document.getElementById('t3');
  if(t1) t1.innerText = new Date(now.getTime()-1000*60*12).toLocaleString();
  if(t2) t2.innerText = new Date(now.getTime()-1000*60*6).toLocaleString();
  if(t3) t3.innerText = new Date(now.getTime()-1000*60*90).toLocaleString();
}
